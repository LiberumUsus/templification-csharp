Simple Button Template

Source Html

<Button>Default Class</Button>
<Button class%o='btn text-blue-700'>Class Change</Button>

Template Html

<style>
 .btn {
  @apply my-2 p-10px rounded-20 cursor-pointer;
  border: 1px solid gray;
 }
</style>

<input class="btn text-white" type="button" value={default}/>
Type Selector Template

Source Html

<Button>Default Class</Button>
<Button type='link'>Link Type</Button>

Template Html

<style>
 .btn {
  @apply my-2 p-10px rounded-20 cursor-pointer;
  border: 1px solid gray;
 }
 .btn-link {
  background: #A7399A;
 }
</style>

{#if type = link}
<a class="btn btn-link">{default}</a>
{:else}
<input class="btn" type="button" value={default}/>
{#endif}
